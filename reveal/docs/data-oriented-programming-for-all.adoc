== Data validation (JSON Schema)

.book-schema.json
[.fragment]
[.centered.w50]
[source,json]
--
{
  "type": "object",
  "required": ["title", "publicationYear"], 
  "properties": {
    "title": {"type": "string"},
    "publicationYear": {"type": "integer"}
  }
}
--

[.left.w50]
--
[.fragment]
Ajv JSON schema validator in JavaScript

[.fragment]
[source,javascript]
var ajv = new Ajv({allErrors: true}); 
var bookSchema = parseJSONFile("book-schema.json");

[.fragment]
[source,javascript]
function bookInfo(book) {
  if(!ajv.validate(bookSchema, book)) {
    throw "bookInfo called with invalid args: " + 
      ajv.errorsText(ajv.errors);
  }
  return book.title + 
    " was published in: "
    + book.publicationYear;
}
